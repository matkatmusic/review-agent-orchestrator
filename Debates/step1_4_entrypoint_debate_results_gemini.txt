import React from 'react';
import { render } from 'ink-testing-library';
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { resetMockData } from './mock-store.js';

// Mock mock-store to verify flag processing without affecting filesystem
vi.mock('./mock-store.js', () => ({
    resetMockData: vi.fn(),
}));

describe('run.tsx entry point', () => {
    const originalArgv = process.argv;
    const originalEnv = process.env.NODE_ENV;

    beforeEach(() => {
        vi.resetModules();
        vi.clearAllMocks();
        process.env.NODE_ENV = 'test';
    });

    afterEach(() => {
        process.argv = originalArgv;
        process.env.NODE_ENV = originalEnv;
    });

    it('--resetMockData flag is processed before render', async () => {
        process.argv = ['node', 'dist/tui/run.js', '--resetMockData'];
        await import('./run.js');
        expect(resetMockData).toHaveBeenCalledOnce();
    });

    it('does not reset mock data without flag', async () => {
        process.argv = ['node', 'dist/tui/run.js'];
        await import('./run.js');
        expect(resetMockData).not.toHaveBeenCalled();

        process.argv = ['node', 'dist/tui/run.js'];
        const { AppWrapper } = await import('./run.js');
        const { lastFrame } = render(<AppWrapper />);
        const frame = lastFrame();
        expect(frame).toContain('Phase 1 Placeholder');
    });
});

=== IMPLEMENTATION FILE: src/tui/run.tsx ===

/**
 * TUI entry point.
 *
 * --resetMockData flag must be processed BEFORE mock-data.ts is imported,
 * so we use dynamic imports for everything that transitively loads mock data.
 */

import { resetMockData } from './mock-store.js';

if (process.argv.includes('--resetMockData')) {
    resetMockData();
    console.log('Mock data reset to defaults.');
}

const [
    { default: React, useState, useEffect, useCallback },
    { render, useStdout, useApp, Box, Text },
    { AppShell },
    { ViewType },
] = await Promise.all([
    import('react'),
    import('ink'),
    import('./app-shell.js'),
    import('./views.js'),
]);

export function AppWrapper() {
    const { stdout } = useStdout();
    const { exit } = useApp();
    const stream = stdout as import('node:tty').WriteStream | undefined;

    const [dims, setDims] = useState({
        columns: stream?.columns ?? 80,
        rows: stream?.rows ?? 24,
    });

    const onResize = useCallback(() => {
        if (stream) {
            setDims({ columns: stream.columns, rows: stream.rows });
        }
    }, [stream]);

    useEffect(() => {
        if (stream) {
            stream.on('resize', onResize);
            return () => { stream.off('resize', onResize); };
        }
    }, [stream, onResize]);

    const currentView = { type: ViewType.Home };

    return (
        <AppShell
            columns={dims.columns}
            rows={dims.rows}
            currentView={currentView}
        >
            {(_setFooterOptions) => (
                <Box justifyContent="center" alignItems="center" flexGrow={1}>
                    <Text>Phase 1 Placeholder</Text>
                </Box>
            )}
        </AppShell>
    );
}

if (process.env.NODE_ENV !== 'test') {
    const instance = render(<AppWrapper />);
    await instance.waitUntilExit();
}
