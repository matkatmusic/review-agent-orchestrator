## 4. is_worktree_merged() — Merge Commit Parent Scan

is_worktree_merged() {
  local branch_name="$1"
  local repo_root="${2:-${REPO_ROOT}}"
  local branch_tip
  branch_tip=$(git -C "${repo_root}" rev-parse "refs/heads/${branch_name}" 2>/dev/null) || return 1

  while IFS= read -r line; do
    read -ra parts <<< "${line}"
    for (( i=2; i<${#parts[@]}; i++ )); do
      [ "${parts[$i]}" = "${branch_tip}" ] && return 0
    done
  done < <(git -C "${repo_root}" log --all --merges --format='%H %P' 2>/dev/null)

  return 1
}

```bash
# worktreeLib.sh: replace merge detection + cleanup with bounded, re-verified logic.
is_worktree_merged() {
  local branch_name="$1" repo_root="$2" birth_iso="$3"
  local branch_tip
  branch_tip="$(git -C "${repo_root}" rev-parse "refs/heads/${branch_name}" 2>/dev/null)" || return 1

  git -C "${repo_root}" log --all --merges --since="${birth_iso}" --format='%P' |
    awk -v tip="${branch_tip}" '
      NF >= 2 {
        for (i = 2; i <= NF; i++) if ($i == tip) found = 1
      }
      END { exit(found ? 0 : 1) }
    '
}

monitor_is_alive() {
  local worktree_path="$1" pid_file="${worktree_path}/.worktree-meta/window-monitor.pid" hb_file="${worktree_path}/.worktree-meta/window-monitor.heartbeat"
  if [[ -f "${pid_file}" ]]; then
    read -r pid < "${pid_file}" || pid=""
    [[ -n "${pid}" ]] && kill -0 "${pid}" 2>/dev/null && return 0
  fi
  if [[ -f "${hb_file}" ]]; then
    read -r last < "${hb_file}" || return 1
    (( $(date +%s) - last < 10 )) && return 0
  fi
  return 1
}

safe_cleanup_worktree() {
  local branch_name="$1" worktree_path="$2" expected_tip="$3"
  local current_tip
  current_tip="$(git -C "${REPO_ROOT}" rev-parse "refs/heads/${branch_name}" 2>/dev/null)" || return 1
  [[ "${current_tip}" == "${expected_tip}" ]] || return 1
  monitor_is_alive "${worktree_path}" && return 1
  if tmux has-session -t "${branch_name}" 2>/dev/null && tmux list-clients -t "${branch_name}" 2>/dev/null | grep -q .; then
    return 1
  fi
  has_non_managed_changes "${worktree_path}" && return 1

  rm -rf "${worktree_path}/.worktree-meta" "${worktree_path}/.worktree.code-workspace"
  git -C "${REPO_ROOT}" worktree remove "${worktree_path}" || return 1
  tmux kill-session -t "${branch_name}" 2>/dev/null || true
  git -C "${REPO_ROOT}" branch -d "${branch_name}" 2>/dev/null || true
  git -C "${REPO_ROOT}" worktree prune >/dev/null 2>&1 || true
}
```

```bash
    CLAUDE_CMD=(claude --permission-mode plan)
  else
    echo "WARNING: claude lacks --permission-mode; starting without plan mode." >&2
    CLAUDE_CMD=(claude)
  fi
}

cd "${WORKTREE_PATH}"
run_npm_install
tmux new-session -A -d -s "${NAME}" -c "${WORKTREE_PATH}"
get_claude_cmd
exec "${CLAUDE_CMD[@]}"
```

```bash
# worktreeLib.sh: replace merge detection + cleanup with bounded, re-verified logic.
is_worktree_merged() {
  local branch_name="$1" repo_root="$2" birth_iso="$3"
  local branch_tip
  branch_tip="$(git -C "${repo_root}" rev-parse "refs/heads/${branch_name}" 2>/dev/null)" || return 1

  git -C "${repo_root}" log --all --merges --since="${birth_iso}" --format='%P' |
    awk -v tip="${branch_tip}" '
      NF >= 2 {
        for (i = 2; i <= NF; i++) if ($i == tip) found = 1
      }
      END { exit(found ? 0 : 1) }
    '
}

monitor_is_alive() {
  local worktree_path="$1" pid_file="${worktree_path}/.worktree-meta/window-monitor.pid" hb_file="${worktree_path}/.worktree-meta/window-monitor.heartbeat"
  if [[ -f "${pid_file}" ]]; then
    read -r pid < "${pid_file}" || pid=""
    [[ -n "${pid}" ]] && kill -0 "${pid}" 2>/dev/null && return 0
  fi
  if [[ -f "${hb_file}" ]]; then
    read -r last < "${hb_file}" || return 1
    (( $(date +%s) - last < 10 )) && return 0
  fi
  return 1
}

safe_cleanup_worktree() {
  local branch_name="$1" worktree_path="$2" expected_tip="$3"
  local current_tip
  current_tip="$(git -C "${REPO_ROOT}" rev-parse "refs/heads/${branch_name}" 2>/dev/null)" || return 1
  [[ "${current_tip}" == "${expected_tip}" ]] || return 1
  monitor_is_alive "${worktree_path}" && return 1
  if tmux has-session -t "${branch_name}" 2>/dev/null && tmux list-clients -t "${branch_name}" 2>/dev/null | grep -q .; then
    return 1
  fi
  has_non_managed_changes "${worktree_path}" && return 1

  rm -rf "${worktree_path}/.worktree-meta" "${worktree_path}/.worktree.code-workspace"
  git -C "${REPO_ROOT}" worktree remove "${worktree_path}" || return 1
  tmux kill-session -t "${branch_name}" 2>/dev/null || true
  git -C "${REPO_ROOT}" branch -d "${branch_name}" 2>/dev/null || true
  git -C "${REPO_ROOT}" worktree prune >/dev/null 2>&1 || true
}
```
